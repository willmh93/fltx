if(TARGET f256_expression_compile_bench)
    return()
endif()

if(NOT DEFINED FLTX_INCLUDE_DIR)
    get_filename_component(FLTX_INCLUDE_DIR "${CMAKE_CURRENT_LIST_DIR}/../../../include" ABSOLUTE)
endif()

if(NOT COMMAND fltx_apply_msvc_timing_reports)
    include("${CMAKE_CURRENT_LIST_DIR}/../../../cmake/fltxMsvcTiming.cmake")
endif()

set(FLTX_F256_EXPRESSION_BENCH_REPEATS "3" CACHE STRING "Number of repeats for each f256 expression compile benchmark case.")
set(FLTX_F256_EXPRESSION_BENCH_COUNT "1000" CACHE STRING "Generated function count for the large f256 expression compile benchmark cases.")

include("${CMAKE_CURRENT_LIST_DIR}/f256_expression_bench_cases.cmake")

set(FLTX_F256_EXPRESSION_BENCH_GENERATED_DIR "${CMAKE_CURRENT_BINARY_DIR}/generated")
file(MAKE_DIRECTORY "${FLTX_F256_EXPRESSION_BENCH_GENERATED_DIR}")

set(FLTX_F256_EXPRESSION_BENCH_GENERATED_SOURCES
    "${FLTX_F256_EXPRESSION_BENCH_GENERATED_DIR}/empty.cpp"
    "${FLTX_F256_EXPRESSION_BENCH_GENERATED_DIR}/inc_f128.cpp"
    "${FLTX_F256_EXPRESSION_BENCH_GENERATED_DIR}/inc_f256.cpp"
    "${FLTX_F256_EXPRESSION_BENCH_GENERATED_DIR}/inc_f128_math.cpp"
    "${FLTX_F256_EXPRESSION_BENCH_GENERATED_DIR}/inc_f256_math.cpp"
    "${FLTX_F256_EXPRESSION_BENCH_GENERATED_DIR}/inc_fltx_math.cpp"
    "${FLTX_F256_EXPRESSION_BENCH_GENERATED_DIR}/inc_f256_io.cpp"
)

foreach(_CASE_ID IN LISTS FLTX_F256_EXPR_BENCH_STRESS_CASES)
    list(APPEND FLTX_F256_EXPRESSION_BENCH_GENERATED_SOURCES
        "${FLTX_F256_EXPRESSION_BENCH_GENERATED_DIR}/expr_${_CASE_ID}_${FLTX_F256_EXPRESSION_BENCH_COUNT}.cpp"
        "${FLTX_F256_EXPRESSION_BENCH_GENERATED_DIR}/eager_${_CASE_ID}_${FLTX_F256_EXPRESSION_BENCH_COUNT}.cpp"
    )
endforeach()

foreach(_GENERATED_SOURCE IN LISTS FLTX_F256_EXPRESSION_BENCH_GENERATED_SOURCES)
    if(NOT EXISTS "${_GENERATED_SOURCE}")
        file(WRITE "${_GENERATED_SOURCE}" "// Generated by f256_expression_compile_bench.\n")
    endif()
endforeach()

add_custom_target(f256_expression_compile_bench
    COMMAND ${CMAKE_COMMAND}
        -DFLTX_F256_EXPR_BENCH_BINARY_DIR=${CMAKE_CURRENT_BINARY_DIR}
        -DFLTX_F256_EXPR_BENCH_INCLUDE_DIR=${FLTX_INCLUDE_DIR}
        -DFLTX_F256_EXPR_BENCH_COMPILER=${CMAKE_CXX_COMPILER}
        -DFLTX_F256_EXPR_BENCH_COMPILER_ID=${CMAKE_CXX_COMPILER_ID}
        -DFLTX_F256_EXPR_BENCH_REPEATS=${FLTX_F256_EXPRESSION_BENCH_REPEATS}
        -DFLTX_F256_EXPR_BENCH_COUNT=${FLTX_F256_EXPRESSION_BENCH_COUNT}
        -P ${CMAKE_CURRENT_LIST_DIR}/run_compile_bench.cmake
    USES_TERMINAL
    VERBATIM
    COMMENT "Timing f256 expression compile cases"
    SOURCES
        "${CMAKE_CURRENT_LIST_DIR}/run_compile_bench.cmake"
        "${CMAKE_CURRENT_LIST_DIR}/f256_expression_bench_cases.cmake"
        ${FLTX_F256_EXPRESSION_BENCH_GENERATED_SOURCES}
)

source_group("scripts" FILES
    "${CMAKE_CURRENT_LIST_DIR}/run_compile_bench.cmake"
    "${CMAKE_CURRENT_LIST_DIR}/f256_expression_bench_cases.cmake"
)
source_group("generated" FILES ${FLTX_F256_EXPRESSION_BENCH_GENERATED_SOURCES})
